Option Explicit

Private Sub cbxNurBestimmteFahrzuege_Click()

End Sub

Private Sub UserForm_Initialize()

    'Breite und Höhe des Popups da VBA es sonst nicht hinkriegt:
    Me.Width = 300
    Me.Height = 200
    'mittige zentrierung des Popups wird ebenfalls durch Code erzwungen:
    Me.StartUpPosition = 0
    Me.Left = application.Left + (application.Width / 2) - (Me.Width / 2)
    Me.Top = application.Top + (application.Height / 2) - (Me.Height / 2)

    Me.Caption = "Baubeschreibungssuche"
    
    Me.txtSuchbegriff.text = "dekor"
    Me.cbxNurBestimmteErgebnisse.value = False
    Me.txtErgebnisFilter.Enabled = False
    Me.txtErgebnisFilter.BackColor = &H8000000F
End Sub

Private Sub cbxNurBestimmteErgebnisse_Click()
    Me.txtErgebnisFilter.Enabled = Me.cbxNurBestimmteErgebnisse.value
    
    If Me.txtErgebnisFilter.Enabled Then
        Me.txtErgebnisFilter.BackColor = vbWhite
    Else
        Me.txtErgebnisFilter.BackColor = &H8000000F
        Me.txtErgebnisFilter.text = ""
    End If
End Sub

Private Sub cmdStart_Click()

    If Len(Trim$(Me.txtBBNummern.text)) = 0 Then
        MsgBox "Bitte mindestens eine BB-Nummer einfügen.", vbExclamation, "Fehlende BB-Nummern"
        Exit Sub
    End If
    
    If Len(Trim$(Me.txtSuchbegriff.text)) = 0 Then
        MsgBox "Bitte einen Suchbegriff eingeben.", vbExclamation, "Fehlender Suchbegriff"
        Exit Sub
    End If
    
    If Me.cbxNurBestimmteErgebnisse.value = True Then
        If Len(Trim$(Me.txtErgebnisFilter.text)) = 0 Then
            MsgBox "Bitte den Ergebnistext für den Filter eingeben oder den Haken entfernen.", vbExclamation, "Fehlender Filtertext"
            Exit Sub
        End If
    End If
    
    gBS_BBInputText = Me.txtBBNummern.text
    gBS_SearchText = Trim$(Me.txtSuchbegriff.text)
    gBS_FilterActive = CBool(Me.cbxNurBestimmteErgebnisse.value)
    gBS_FilterText = Trim$(Me.txtErgebnisFilter.text)
    gBS_FormConfirmed = True
    
    Me.Hide
End Sub

Private Sub UserForm_QueryClose(Cancel As Integer, CloseMode As Integer)
    gBS_FormConfirmed = False
End Sub

Private Sub UserForm_Activate()
    BS_MakeFormTopMost Me.Caption
End Sub
